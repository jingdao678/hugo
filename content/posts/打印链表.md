```python
class Node:
    def __init__(self, data):
        self.data = data
        self.next = None


def print_linked_list(head):
    current = head

    while current is not None:
        print(current.data)
        current = current.next


# 创建链表节点
node1 = Node(9)
node2 = Node(2)
node3 = Node(3)

# 构建链表关系
node1.next = node2
node2.next = node3

# 打印链表
print_linked_list(node1)
```
